class etckeeper::yum (
  Enum['present', 'absent'] $ensure = 'present',
  Boolean                   $enable = true,
) {

  if ! defined(Class['::etckeeper']) {
    fail('You must include the etckeeper base class before using the etckeeper::yum class')
  }

  if $ensure == 'present' {
    ::yum::plugin { 'etckeeper':
      content => template("${module_name}/yum.conf.erb"),
    }
  }
}
